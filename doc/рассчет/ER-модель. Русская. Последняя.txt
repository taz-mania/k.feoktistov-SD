    }

        uuid идентификатор_события PK
        uuid идентификатор_резерва
        string тип_события
    }
    
    %% ==================== Мастер-данные ====================
    КАТЕГОРИИ_ТОВАРОВ {
        string код_категории PK
        string наименование
        string код_родителя FK
    }

        string артикул PK
        string наименование
        string код_категории FK
    }
    
    МАГАЗИНЫ {
        string код_магазина PK
        string наименование
        string адрес

        string код_магазина FK
        boolean активен

        uuid идентификатор PK
        string код_магазина FK
        decimal широта
        decimal долгота
    }
    
    %% ==================== Управление доступом ====================
    ПОЛЬЗОВАТЕЛИ {
        uuid идентификатор PK
        string логин
    }
    
    РОЛИ_ДОСТУПА {
        int код_роли PK
        string наименование_роли
    }
    
    НАЗНАЧЕНИЕ_РОЛЕЙ {
        uuid идентификатор_пользователя FK
        int код_роли FK
    }
    
    %% ==================== Резервирование ====================
    РЕЗЕРВЫ {
        uuid номер_резерва PK
        string номер_заказа
        string статус
    }
    
    СТАТУСЫ_РЕЗЕРВОВ {
        string код_статуса PK
        string наименование
    }
    
    ПОЗИЦИИ_РЕЗЕРВА {
        uuid идентификатор_позиции PK
        uuid номер_резерва FK
        string артикул
        string код_магазина
    }
    
    ЛИМИТЫ_ПОЛЬЗОВАТЕЛЕЙ {
        string идентификатор_пользователя PK
        int количество_активных
    }
    
    ИСХОДЯЩИЕ_СОБЫТИЯ {
        uuid идентификатор PK
        uuid номер_резерва
        string тип_события
    }
    
    %% ==================== Read Models ====================
    СРЕЗЫ_ОСТАТКОВ {
        string ключ_среза PK
        string артикул
        string код_магазина
        int доступное_количество
    }
    
    %% ==================== Аналитика ====================
    ВРЕМЕННЫЕ_РЯДЫ {
        timestamp время PK
        string артикул PK
        string код_магазина PK
        string метрика PK
        double значение
    }
    
    %% ==================== Конфликты ====================
    КОНФЛИКТЫ_ДАННЫХ {
        uuid номер_конфликта PK
        string артикул
        string код_магазина
        string статус
    }
    
    ПРАВИЛА_СВЕРКИ {
        int идентификатор_правила PK
        string код_категории
        int порог_абсолютный
    }
